Respostas

1 - Escreva um programa que exiba a mensagem "Olá, Mundo!" na tela.

      
      #include< stdio.h >  
      #include< stdlib.h >
      
      int main() { 

         // Exibe a mensagem 
         printf("Olá, Mundo!");  
         return 0; 
      };

2 - Escreva um programa que leia um número inteiro digitado pelo usuário e o exiba de volta.

      
      #include< stdio.h >  
      #include< stdlib.h >
      int main() { 
         // Variáveis
         int numero; 
      
         // Entrada de dados 
         printf("Digite um numero inteiro: "); 
         scanf("%i", &numero);
         
         // Saída de dados 
         printf("Voce digitou o numero: %i", numero);  

         return 0;
      };
